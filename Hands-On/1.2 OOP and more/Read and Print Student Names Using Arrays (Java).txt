import java.util.Scanner;

public class Main {
	public static void main(String[] args) {
		Scanner in = new Scanner(System.in);
		// Read number of students
		int n = in.nextInt();
		
		// TODO: Create an array to store student names
		String[] students = new String[n];
		// TODO: Read n student names into the array
		for (int i = 0; i < n; i++){
			students[i] = in.next();
		}
		// TODO: Print each student name on a new line in input order
		for (String student : students){
			System.out.println(student);
		}
		
		in.close();
	}
}
