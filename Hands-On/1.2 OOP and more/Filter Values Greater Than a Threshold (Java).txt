import java.util.Scanner;

public class Main {

    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        // Read number of values
        int n = in.nextInt();

        // Read input values
        int[] values = new int[n];
        for (int i = 0; i < n; i++) {
            values[i] = in.nextInt();
        }

        // Read threshold
        int threshold = in.nextInt();

        // Create NumberFilter object
        NumberFilter filter = new NumberFilter(values);

        // Filter values greater than threshold
        int[] result = filter.filterGreaterThan(threshold);

        // Print result
        if (result.length == 0) {
            System.out.println("No values above threshold");
        } else {
            for (int i = 0; i < result.length; i++) {
                System.out.print(result[i]);
                if (i < result.length - 1) {
                    System.out.print(" ");
                }
            }
        }

        in.close();
    }
}

// TODO: Implement this class
class NumberFilter {

    private int[] numbers;

    // Constructor
    public NumberFilter(int[] numbers) {
        this.numbers = numbers;
    }

    // TODO: Implement this method
    // Returns an array of values strictly greater than threshold
    // Preserves original order
    public int[] filterGreaterThan(int threshold) {

       int[] filtered = new int[numbers.length];
       int count = 0;

       for (int number : numbers){
            if (number > threshold){
                filtered[count] = number;
                count++;
            }
       }
        int[] result = new int[count];
        for (int i = 0; i < count; i++){
            result[i] = filtered[i];
        }
        return result; 
    }
}
