import java.util.*;
import java.util.stream.*;

public class Main {

    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        // Read number of threat readings
        int n = in.nextInt();

        // Read threat readings
        List<Integer> threats = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            threats.add(in.nextInt());
        }

        // Read danger threshold
        int threshold = in.nextInt();

        // Create ThreatAnalyzer object
        ThreatAnalyzer analyzer = new ThreatAnalyzer(threats);

        // Analyze threats
        List<String> alerts = analyzer.analyzeThreats(threshold);

        // Print results
        if (alerts.isEmpty()) {
            System.out.println("No threats detected");
        } else {
            for (String alert : alerts) {
                System.out.println(alert);
            }
        }

        in.close();
    }
}

// TODO: Implement this class
class ThreatAnalyzer {

    private List<Integer> threats;

    // Constructor
    public ThreatAnalyzer(List<Integer> threats) {
        this.threats = threats;
    }

    // TODO: Implement this method using Java Streams
    // - Filter threat readings strictly greater than threshold
    // - Map each reading to: "HIGH THREAT: x"
    // - Preserve original order
    // - Return a list of alert strings
    public List<String> analyzeThreats(int threshold) {

        // TODO: Write Stream logic here

        return threats.stream()
                .filter(threat -> threat > threshold)
                .map(threat -> "HIGH THREAT: " + threat)
                .collect(Collectors.toList());
    }
}
