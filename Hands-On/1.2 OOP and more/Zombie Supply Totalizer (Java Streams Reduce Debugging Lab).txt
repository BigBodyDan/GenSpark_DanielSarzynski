import java.util.*;

public class Main {

    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        int n = in.nextInt();

        List<Integer> crates = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            crates.add(in.nextInt());
        }

        SupplyAggregator aggregator = new SupplyAggregator(crates);
        int total = aggregator.totalSupplies();

        System.out.println(total);

        in.close();
    }
}
