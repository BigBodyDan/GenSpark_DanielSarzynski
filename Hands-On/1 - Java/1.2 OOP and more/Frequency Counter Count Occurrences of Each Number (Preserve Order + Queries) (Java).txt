import java.util.Scanner;
import java.util.Map;
import java.util.LinkedHashMap;

public class Main {

    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        // Read number of elements
        int n = in.nextInt();

        // Read array elements
        int[] numbers = new int[n];
        for (int i = 0; i < n; i++) {
            numbers[i] = in.nextInt();
        }

        // Build frequency map
        Map<Integer, Integer> frequencyMap = buildFrequencyMap(numbers);

        // Read number of queries
        int q = in.nextInt();

        // Process queries
        for (int i = 0; i < q; i++) {
            int query = in.nextInt();
            int count = frequencyMap.getOrDefault(query, 0);
            System.out.println(query + ": " + count);
        }

        in.close();
    }

    // TODO: Implement this method
    // Builds a frequency map while preserving order of first appearance
    public static Map<Integer, Integer> buildFrequencyMap(int[] arr) {

        Map<Integer, Integer> frequencyMap = new LinkedHashMap<>();

        for (int num : arr){
            if (frequencyMap.containsKey(num)){
                frequencyMap.put(num, frequencyMap.get(num) + 1);
            } else{
                frequencyMap.put(num, 1);
            }
        }
        return frequencyMap; 
    }
}
