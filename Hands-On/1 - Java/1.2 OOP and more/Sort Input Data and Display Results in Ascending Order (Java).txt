import java.util.*;

public class Main {

    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        // Read number of elements
        int n = in.nextInt();

        // Read input array
        int[] numbers = readArray(in, n);

        // Sort the array
        sortArray(numbers);

        // Print sorted result
        printArray(numbers);

        in.close();
    }

    // TODO: Read n integers and return them as an array
    public static int[] readArray(Scanner in, int n) {
        int[] arr = new int[n];

        for (int i = 0; i < n; i++){
            arr[i] = in.nextInt();
        }

        return arr;
    }

    // TODO: Sort the array in ascending order
    public static void sortArray(int[] arr) {

        Arrays.sort(arr);
    }

    // TODO: Print array elements on one line separated by spaces
    public static void printArray(int[] arr) {

        for (int i = 0; i < arr.length; i++){
            System.out.print(arr[i] + " ");
        }
    }
}
