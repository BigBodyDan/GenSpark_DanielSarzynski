import java.util.Scanner;

public class Main {
	public static void main(String[] args) {

    Scanner in = new Scanner(System.in);

    // Read inputs
    String name = in.next();
    int a = in.nextInt();
    int b = in.nextInt();
    int n = in.nextInt();

    // TODO: Call printGreeting(name)
    printGreeting(name);

    // TODO: Call addNumbers(a, b), capture the result, and print it
    int result = addNumbers(a, b);
    System.out.println("Sum: " + result);

    // TODO: Call isEven(n), capture the result, and print n is even. or n is odd.
    boolean evenOrOdd = isEven(n);
    if(evenOrOdd){
        System.out.println(n + " is even.");
    }
    else{
        System.out.println(n + " is odd.");
    }

    in.close();
}

    // TODO: implement printGreeting(String name)
    // Prints: Hello, <name>!
    public static void printGreeting(String name){
        System.out.println("Hello, " + name + "!");
    }

	// TODO: implement addNumbers(int a, int b)
	// Returns the sum of a and b
    public static int addNumbers(int a, int b){
        return a + b;
    }

	// TODO: implement isEven(int n)
	// Returns true if n is even, false otherwise
    public static boolean isEven(int n){
        return (n % 2 == 0) ? true : false;
    }
}
