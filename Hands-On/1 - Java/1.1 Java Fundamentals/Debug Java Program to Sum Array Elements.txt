import java.util.Scanner;

public class Main {
	public static void main(String[] args) {
		Scanner in = new Scanner(System.in);

    // Read number of elements
    int N = in.nextInt();

    int[] numbers = new int[N];

    // BUG 1: Off-by-one error in loop condition
    for (int i = 0; i < N; i++) {
        numbers[i] = in.nextInt();
    }

    // BUG 2: Incorrect initial value for sum
    int sum = 0;

    // BUG 3: Loop does not process all elements correctly
    for (int i = 0; i < N; i++) {
        sum = sum + numbers[i];
    }

    // BUG 4: Output formatting is correct, but value is wrong
    System.out.println(sum);

    in.close();
	}
}
